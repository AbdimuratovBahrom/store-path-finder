        function updateRows() {
            var block = $('#block').val();
            $.get('/get_rows', { block: block }, function (data) {
                $('#row').empty().append('<option value="">Выберите ряд</option>');
                $.each(data, function (index, row) {
                    $('#row').append('<option value="' + row + '">' + row + '</option>');
                });
                $('#store_name').empty().append('<option value="">Выберите магазин</option>');
            });
        }

        function updateStores() {
            var block = $('#block').val();
            var row = $('#row').val();
            $.get('/get_stores', { block: block, row: row }, function (data) {
                $('#store_name').empty().append('<option value="">Выберите магазин</option>');
                $.each(data, function (index, store) {
                    $('#store_name').append('<option value="' + store + '">' + store + '</option>');
                });
            });
        }

        $(document).ready(function () {
            $('form').submit(function (event) {
                event.preventDefault();
                $.post('/get_path', $(this).serialize(), function (data) {
                    $('.result').html('<h2>Результат:</h2><p>' + data.path + '</p>');
                });
            });
        });
